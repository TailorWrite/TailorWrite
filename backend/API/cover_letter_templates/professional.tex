\newenvironment{twocolentry}[2][]{
    \onecolentry
    \def\secondColumn{#2}
    \setcolumnwidth{\fill, 0.4\textwidth}
    \begin{paracol}{2}
}{
    \switchcolumn \raggedleft \secondColumn
    \end{paracol}
    \endonecolentry
} % new environment for two column entries

\newenvironment{threecolentry}[3][]{
    \onecolentry
    \def\secondColumn{#2}
    \def\thirdColumn{#3}
    \setcolumnwidth{\fill, 4.8 cm, 6.5 cm} % Adjust these widths as needed
    \begin{paracol}{3}
}{
    \switchcolumn[1] \raggedleft \secondColumn
    \switchcolumn[2] \raggedleft \thirdColumn
    \end{paracol}
    \endonecolentry
} % new environment for three column entries
